Mirror Maze

发表于2025-01-23|更新于2025-01-23
### 题目大意 题目描述了一个由 `n × m` 的矩形镜子组成的迷宫，镜子有四种类型：`\`, `/`, `-`, `|`。每种镜子会根据光线的入射方向反射光线。题目中有 `q` 个询问，每个询问给定一个点光源的位置 `(x, y)` 和光线的方向 `dir`，要求计算这束光线在经过足够时间后被多少个不同的镜子反射过。 ### 题解 1. **光路性质**：由于光路是可逆的，光线在镜子迷宫中不会分叉或汇合，因此所有的光线路径要么形成环，要么形成链。 2. **预处理**：我们需要预处理所有可能的光线路径，并记录每条路径上被反射的镜子数量。对于链状路径，光线最终会离开迷宫；对于环状路径，光线会在环中无限循环。 3. **DFS模拟**：通过深度优先搜索（DFS）模拟光线的反射过程，记录每条路径上经过的镜子。对于链状路径，从起点开始模拟直到光线离开迷宫；对于环状路径，模拟直到回到起点。 4. **查询处理**：对于每个查询，根据光线的起点和方向，直接查表得到答案。 ### 代码实现 ```cpp #include #define pii pair #define tii tuple #define all(a) a.begin(), a.end() using namespace std; const int maxn = 1e3 + 10; char ch[maxn][maxn]; int n, m, ans[maxn][maxn][4]; int vis[maxn][maxn][4]; vector vec; // 存储经过了哪些镜子的哪些状态 // 判断某个状态是否会经过镜子的反射 bool istan(int x, int y, int dir) { char tem = ch[x][y]; if (tem == '/' || tem == '\\') return 1; if (tem == '-' && (dir == 0 || dir == 1)) return 1; if (tem == '|' && (dir == 2 || dir == 3)) return 1; return 0; } // 模拟光线的反射过程 void dfs(int x, int y, int dir) { if (x < 1 || x > n || y < 1 || y > m) return; // 链的退出条件 if (vis[x][y][dir]) return; // 环的退出条件 vec.push_back(tii(x, y, dir)); vis[x][y][dir] = 1; if (ch[x][y] == '/') { if (dir == 0) dfs(x, y + 1, 3); else if (dir == 1) dfs(x, y - 1, 2); else if (dir == 2) dfs(x + 1, y, 1); else dfs(x - 1, y, 0); } else if (ch[x][y] == '\\') { if (dir == 0) dfs(x, y - 1, 2); else if (dir == 1) dfs(x, y + 1, 3); else if (dir == 2) dfs(x - 1, y, 0); else dfs(x + 1, y, 1); } else if (ch[x][y] == '-') { if (dir == 0) dfs(x + 1, y, 1); else if (dir == 1) dfs(x - 1, y, 0); else if (dir == 2) dfs(x, y - 1, 2); else dfs(x, y + 1, 3); } else if (ch[x][y] == '|') { if (dir == 0) dfs(x - 1, y, 0); else if (dir == 1) dfs(x + 1, y, 1); else if (dir == 2) dfs(x, y + 1, 3); else dfs(x, y - 1, 2); } } // 处理链状路径 void solve1(int x, int y, int dir) { vec.clear(); dfs(x, y, dir); reverse(all(vec)); // 倒叙遍历状态更新答案 set s; for (auto [a, b, c] : vec) { if (istan(a, b, c)) s.insert(pii(a, b)); ans[a][b][c] = s.size(); } } // 处理环状路径 void solve2(int x, int y, int dir) { vec.clear(); dfs(x, y, dir); set s; for (auto [a, b, c] : vec) { if (istan(a, b, c)) s.insert(pii(a, b)); } for (auto [a, b, c] : vec) ans[a][b][c] = s.size(); // 环上的答案都一样 } int main() { ios::sync_with_stdio(false); cin.tie(0), cout.tie(0); cin >> n >> m; for (int i = 1; i <= n; i++) for (int j = 1; j <= m; j++) cin >> ch[i][j]; // 0，1，2，3分别表示为向上、下、左、右射入的光线 // 从边界射入的光线一定是链 for (int i = 1; i <= n; i++) solve1(i, 1, 3), solve1(i, m, 2); for (int j = 1; j <= m; j++) solve1(1, j, 1), solve1(n, j, 0); // 还未访问的状态，一定是环上的状态 for (int i = 1; i <= n; i++) for (int j = 1; j <= m; j++) for (int k = 0; k < 4; k++) if (!vis[i][j][k]) solve2(i, j, k); int q; cin >> q; map mp; mp["above"] = 0; mp["below"] = 1; mp["left"] = 2; mp["right"] = 3; while (q--) { int x, y; string str; cin >> x >> y >> str; // 先根据方向偏移一格 int tem = mp[str]; if (tem == 0) x--; else if (tem == 1) x++; else if (tem == 2) y--; else y++; cout << ans[x][y][tem] << "\n"; } return 0; } ``` ### 代码解释 1. **DFS模拟**：`dfs` 函数模拟光线的反射过程，根据镜子的类型和光线的方向决定光线的下一步走向。 2. **链状路径处理**：`solve1` 函数处理链状路径，从起点开始模拟光线反射，记录经过的镜子，并更新答案。 3. **环状路径处理**：`solve2` 函数处理环状路径，模拟光线反射直到回到起点，记录经过的镜子，并更新答案。 4. **查询处理**：对于每个查询，根据光线的起点和方向，直接查表得到答案。 ### 时间复杂度 - **预处理**：`O(nm)`，因为每个状态最多被访问一次。 - **查询处理**：`O(q)`，每个查询只需要查表即可。 总时间复杂度为 `O(nm + q)`，能够满足题目要求。
